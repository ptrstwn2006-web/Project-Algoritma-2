DEFINE CONSTANT JUMLAH_CALON = 3

DEFINE STRUCT Calon:
    nama: string
    suara: integer

MAIN PROGRAM:
    DEKLARASI array dpr[JUMLAH_CALON] bertipe Calon
    INISIALISASI:
        dpr[0].nama = "Calon A", dpr[0].suara = 0
        dpr[1].nama = "Calon B", dpr[1].suara = 0
        dpr[2].nama = "Calon C", dpr[2].suara = 0

    PRINT "Masukkan jumlah pemilih"
    INPUT jumlahPemilih

    FOR i FROM 1 TO jumlahPemilih DO:
        PRINT daftar calon (1. Calon A, 2. Calon B, 3. Calon C)
        PRINT "Pilihan Anda (1-3):"
        INPUT pilihan

        IF pilihan >= 1 AND pilihan <= JUMLAH_CALON THEN
            INCREMENT dpr[pilihan - 1].suara
        ELSE
            PRINT "Pilihan tidak valid. Suara tidak dihitung."
        ENDIF
    ENDFOR

    PRINT "=== Hasil Voting ==="
    FOR i FROM 0 TO JUMLAH_CALON - 1 DO:
        PRINT dpr[i].nama, dpr[i].suara
    ENDFOR

    // Menentukan pemenang
    SET maxSuara = dpr[0].suara
    SET indeksPemenang = 0

    FOR i FROM 1 TO JUMLAH_CALON - 1 DO:
        IF dpr[i].suara > maxSuara THEN
            SET maxSuara = dpr[i].suara
            SET indeksPemenang = i
        ENDIF
    ENDFOR

    PRINT "Pemenang adalah: ", dpr[indeksPemenang].nama, " dengan ", maxSuara, " suara"

END PROGRAM
